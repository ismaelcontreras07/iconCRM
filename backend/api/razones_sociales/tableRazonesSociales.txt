
CREATE TABLE IF NOT EXISTS `razones_sociales` (
  `id`           INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `account_id`   INT UNSIGNED NOT NULL,              -- FK a accounts.id
  `nombre`       VARCHAR(255)       NOT NULL,        -- la raz√≥n social
  `tipo`         ENUM('principal','secundaria') 
                   NOT NULL DEFAULT 'principal',     -- opcional para diferenciar
  `status`       ENUM('activo','inactivo') 
                   NOT NULL DEFAULT 'activo',
  `created_at`   TIMESTAMP          NOT NULL 
                   DEFAULT CURRENT_TIMESTAMP,
  `updated_at`   TIMESTAMP          NOT NULL 
                   DEFAULT CURRENT_TIMESTAMP 
                   ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  INDEX `idx_rs_account` (`account_id`),
  CONSTRAINT `fk_rs_account`
    FOREIGN KEY (`account_id`) REFERENCES `accounts`(`id`)
      ON DELETE CASCADE
      ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
