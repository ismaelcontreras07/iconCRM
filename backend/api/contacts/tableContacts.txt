CREATE TABLE `contacts` (
  `id` INT UNSIGNED NOT NULL AUTO_INCREMENT,
  `account_id` INT UNSIGNED         NOT NULL,              -- Relaci√≥n a accounts.id
  `first_name` VARCHAR(100)         NOT NULL,
  `last_name` VARCHAR(100)          NOT NULL,
  `title` VARCHAR(100)              DEFAULT NULL,          -- Cargo o puesto
  `department` VARCHAR(100)         DEFAULT NULL,
  `email` VARCHAR(150)              DEFAULT NULL,
  `phone` VARCHAR(50)               DEFAULT NULL,
  `mobile` VARCHAR(50)              DEFAULT NULL,
  `lead_source` VARCHAR(100)        DEFAULT NULL,          -- Ej. web, evento, referencia
  `status` ENUM('activo','inactivo') NOT NULL DEFAULT 'activo',
  `created_at` TIMESTAMP            NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` TIMESTAMP            NOT NULL 
      DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  INDEX `idx_contacts_account` (`account_id`),
  CONSTRAINT `fk_contacts_account`
    FOREIGN KEY (`account_id`) REFERENCES `accounts`(`id`)
      ON DELETE CASCADE
      ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
